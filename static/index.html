Lesson 1: Variables & Data Types

Introduction to Variables

In Go, variables are used to store data. Go is a statically typed language, which means you need to declare the type of a variable before using it (though type inference makes this easier).

Declaring Variables

There are several ways to declare variables in Go:

package main

import "fmt"

func main() {
    // Method 1: var keyword with explicit type
    var name string = "John"
    var age int = 25
    
    // Method 2: var keyword with type inference
    var city = "New York"
    var temperature = 72.5
    
    // Method 3: Short declaration (inside functions only)
    country := "USA"
    isStudent := true
    
    fmt.Println("Name:", name)
    fmt.Println("Age:", age)
    fmt.Println("City:", city)
    fmt.Println("Temperature:", temperature)
    fmt.Println("Country:", country)
    fmt.Println("Is Student:", isStudent)
}

Basic Data Types

Go has several built-in data types:

  • int - Integer numbers (int8, int16, int32, int64)
  • uint - Unsigned integers (uint8, uint16, uint32, uint64)
  • float32, float64 - Floating-point numbers
  • string - Text data
  • bool - Boolean (true or false)
  • byte - Alias for uint8
  • rune - Alias for int32 (represents a Unicode character)
package main

import "fmt"

func main() {
    // Integer types
    var smallNumber int8 = 127
    var bigNumber int64 = 9223372036854775807
    
    // Floating-point types
    var pi float32 = 3.14
    var precisePi float64 = 3.14159265359
    
    // String type
    var greeting string = "Hello, Go!"
    
    // Boolean type
    var isGoPowerful bool = true
    
    // Byte and Rune
    var firstLetter byte = 'A'
    var unicodeChar rune = 'ä¸–'
    
    fmt.Printf("Small Number: %d (type: %T)\n", smallNumber, smallNumber)
    fmt.Printf("Big Number: %d (type: %T)\n", bigNumber, bigNumber)
    fmt.Printf("Pi: %.2f (type: %T)\n", pi, pi)
    fmt.Printf("Precise Pi: %.11f (type: %T)\n", precisePi, precisePi)
    fmt.Printf("Greeting: %s (type: %T)\n", greeting, greeting)
    fmt.Printf("Is Go Powerful: %t (type: %T)\n", isGoPowerful, isGoPowerful)
    fmt.Printf("First Letter: %c (type: %T)\n", firstLetter, firstLetter)
    fmt.Printf("Unicode Char: %c (type: %T)\n", unicodeChar, unicodeChar)
}

Zero Values

In Go, variables declared without an initial value are given their zero value:

package main

import "fmt"

func main() {
    var i int        // 0
    var f float64    // 0.0
    var b bool       // false
    var s string     // "" (empty string)
    
    fmt.Printf("int: %d\n", i)
    fmt.Printf("float64: %f\n", f)
    fmt.Printf("bool: %t\n", b)
    fmt.Printf("string: %q\n", s)
}
ðŸ’¡ Tip: The short declaration operator := is very convenient but can only be used inside functions. Use var for package-level variables.

Multiple Variable Declaration

package main

import "fmt"

func main() {
    // Declare multiple variables of the same type
    var x, y, z int = 1, 2, 3
    
    // Multiple variables with type inference
    var name, age = "Alice", 30
    
    // Short declaration with multiple variables
    firstName, lastName := "Bob", "Smith"
    
    fmt.Println(x, y, z)
    fmt.Println(name, age)
    fmt.Println(firstName, lastName)
}

Lesson 2: Constants & Control Structures

Constants

Constants are declared like variables but with the const keyword. Their values cannot be changed after declaration.

package main

import "fmt"

func main() {
    const Pi = 3.14159
    const AppName = "Go Learning App"
    const MaxUsers = 100
    
    // Multiple constants
    const (
        StatusOK = 200
        StatusNotFound = 404
        StatusError = 500
    )
    
    fmt.Println("Pi:", Pi)
    fmt.Println("App Name:", AppName)
    fmt.Println("Max Users:", MaxUsers)
    fmt.Println("Status OK:", StatusOK)
}

If-Else Statements

The if statement in Go is similar to other languages but doesn't require parentheses around the condition:

package main

import "fmt"

func main() {
    age := 18
    
    if age >= 18 {
        fmt.Println("You are an adult")
    } else {
        fmt.Println("You are a minor")
    }
    
    // If with a short statement
    if score := 85; score >= 90 {
        fmt.Println("Grade: A")
    } else if score >= 80 {
        fmt.Println("Grade: B")
    } else if score >= 70 {
        fmt.Println("Grade: C")
    } else {
        fmt.Println("Grade: F")
    }
}

Switch Statements

Go's switch is more powerful than in many other languages. It doesn't require a break statement:

package main

import (
    "fmt"
    "time"
)

func main() {
    // Basic switch
    day := "Monday"
    
    switch day {
    case "Monday":
        fmt.Println("Start of the work week")
    case "Friday":
        fmt.Println("Almost weekend!")
    case "Saturday", "Sunday":
        fmt.Println("Weekend!")
    default:
        fmt.Println("Midweek day")
    }
    
    // Switch without a condition (like if-else chain)
    hour := time.Now().Hour()
    
    switch {
    case hour < 12:
        fmt.Println("Good morning!")
    case hour < 17:
        fmt.Println("Good afternoon!")
    default:
        fmt.Println("Good evening!")
    }
    
    // Switch with type checking
    var i interface{} = "hello"
    
    switch v := i.(type) {
    case int:
        fmt.Printf("Integer: %d\n", v)
    case string:
        fmt.Printf("String: %s\n", v)
    case bool:
        fmt.Printf("Boolean: %t\n", v)
    default:
        fmt.Printf("Unknown type\n")
    }
}

For Loops

Go has only one looping construct: the for loop. However, it's very flexible:

package main

import "fmt"

func main() {
    // Classic for loop
    for i := 0; i < 5; i++ {
        fmt.Println("Count:", i)
    }
    
    // While-style loop
    counter := 0
    for counter < 3 {
        fmt.Println("Counter:", counter)
        counter++
    }
    
    // Infinite loop (with break)
    num := 0
    for {
        num++
        if num > 3 {
            break
        }
        fmt.Println("Num:", num)
    }
    
    // Continue statement
    for i := 0; i < 5; i++ {
        if i == 2 {
            continue
        }
        fmt.Println("i:", i)
    }
    
    // Range over array/slice
    numbers := []int{10, 20, 30, 40, 50}
    for index, value := range numbers {
        fmt.Printf("Index: %d, Value: %d\n", index, value)
    }
    
    // Range with just values
    for _, value := range numbers {
        fmt.Println("Value:", value)
    }
}
Key Difference: Unlike C or Java, Go's for loop doesn't use parentheses around the three components. Braces are always required, even for single-line loops.

Lesson 3: Functions & Methods

Function Basics

Functions are the building blocks of Go programs. They can take parameters and return values.

package main

import "fmt"

// Function with no parameters and no return value
func sayHello() {
    fmt.Println("Hello, World!")
}

// Function with parameters
func greet(name string) {
    fmt.Printf("Hello, %s!\n", name)
}

// Function with parameters and return value
func add(a int, b int) int {
    return a + b
}

// Function with shortened parameter declaration
func multiply(a, b int) int {
    return a * b
}

func main() {
    sayHello()
    greet("Alice")
    
    sum := add(5, 3)
    fmt.Println("Sum:", sum)
    
    product := multiply(4, 7)
    fmt.Println("Product:", product)
}

Multiple Return Values

Go functions can return multiple values, which is commonly used for error handling:

package main

import (
    "fmt"
    "errors"
)

// Function returning multiple values
func divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, errors.New("division by zero")
    }
    return a / b, nil
}

// Named return values
func rectangle(length, width float64) (area, perimeter float64) {
    area = length * width
    perimeter = 2 * (length + width)
    return // naked return
}

func main() {
    // Handle multiple return values
    result, err := divide(10, 2)
    if err != nil {
        fmt.Println("Error:", err)
    } else {
        fmt.Println("Result:", result)
    }
    
    // Trying to divide by zero
    _, err = divide(10, 0)
    if err != nil {
        fmt.Println("Error:", err)
    }
    
    // Named return values
    a, p := rectangle(5, 3)
    fmt.Printf("Area: %.2f, Perimeter: %.2f\n", a, p)
}

Variadic Functions

Functions can accept a variable number of arguments using the ... syntax:

package main

import "fmt"

// Variadic function
func sum(numbers ...int) int {
    total := 0
    for _, num := range numbers {
        total += num
    }
    return total
}

func printAll(prefix string, items ...string) {
    fmt.Println(prefix)
    for _, item := range items {
        fmt.Println("-", item)
    }
}

func main() {
    fmt.Println("Sum of 1, 2, 3:", sum(1, 2, 3))
    fmt.Println("Sum of 5, 10, 15, 20:", sum(5, 10, 15, 20))
    
    printAll("Fruits:", "Apple", "Banana", "Orange")
    printAll("Colors:", "Red", "Green", "Blue", "Yellow")
}

Anonymous Functions & Closures

Go supports anonymous functions which can form closures:

package main

import "fmt"

func main() {
    // Anonymous function
    func() {
        fmt.Println("This is an anonymous function")
    }()
    
    // Assign anonymous function to variable
    square := func(x int) int {
        return x * x
    }
    fmt.Println("Square of 5:", square(5))
    
    // Closure: function that references variables from outside its body
    counter := func() func() int {
        count := 0
        return func() int {
            count++
            return count
        }
    }()
    
    fmt.Println(counter()) // 1
    fmt.Println(counter()) // 2
    fmt.Println(counter()) // 3
}
ðŸ’¡ Tip: Variadic parameters must be the last parameter in the function signature. You can pass a slice to a variadic function using the ... operator: sum(mySlice...)

Lesson 4: Structs, Interfaces & OOP

Structs

Go doesn't have classes, but it has structs. A struct is a collection of fields that group data together:

package main

import "fmt"

// Define a struct
type Person struct {
    FirstName string
    LastName  string
    Age       int
}

func main() {
    // Create struct instances
    person1 := Person{
        FirstName: "John",
        LastName:  "Doe",
        Age:       30,
    }
    
    // Short form
    person2 := Person{"Jane", "Smith", 25}
    
    // Access fields
    fmt.Println("Person 1:", person1.FirstName, person1.LastName)
    fmt.Println("Person 2:", person2.FirstName, person2.LastName)
    
    // Modify fields
    person1.Age = 31
    fmt.Println("Updated Age:", person1.Age)
    
    // Pointer to struct
    person3 := &Person{
        FirstName: "Bob",
        LastName:  "Johnson",
        Age:       40,
    }
    fmt.Println("Person 3:", person3.FirstName)
}

Methods

Methods are functions with a special receiver argument. This is how Go implements object-oriented behavior:

package main

import "fmt"

type Rectangle struct {
    Width  float64
    Height float64
}

// Method with value receiver
func (r Rectangle) Area() float64 {
    return r.Width * r.Height
}

// Method with pointer receiver (can modify the struct)
func (r *Rectangle) Scale(factor float64) {
    r.Width *= factor
    r.Height *= factor
}

// Method that returns a formatted string
func (r Rectangle) String() string {
    return fmt.Sprintf("Rectangle(%.2f x %.2f)", r.Width, r.Height)
}

func main() {
    rect := Rectangle{Width: 10, Height: 5}
    
    fmt.Println("Area:", rect.Area())
    fmt.Println("Before scaling:", rect)
    
    rect.Scale(2)
    fmt.Println("After scaling:", rect)
    fmt.Println("New area:", rect.Area())
}

Interfaces

Interfaces define behavior. A type implements an interface by implementing its methods (implicitly, no "implements" keyword):

package main

import (
    "fmt"
    "math"
)

// Define an interface
type Shape interface {
    Area() float64
    Perimeter() float64
}

// Rectangle type
type Rectangle struct {
    Width, Height float64
}

func (r Rectangle) Area() float64 {
    return r.Width * r.Height
}

func (r Rectangle) Perimeter() float64 {
    return 2 * (r.Width + r.Height)
}

// Circle type
type Circle struct {
    Radius float64
}

func (c Circle) Area() float64 {
    return math.Pi * c.Radius * c.Radius
}

func (c Circle) Perimeter() float64 {
    return 2 * math.Pi * c.Radius
}

// Function that accepts any Shape
func printShapeInfo(s Shape) {
    fmt.Printf("Area: %.2f, Perimeter: %.2f\n", s.Area(), s.Perimeter())
}

func main() {
    rect := Rectangle{Width: 10, Height: 5}
    circle := Circle{Radius: 7}
    
    fmt.Println("Rectangle:")
    printShapeInfo(rect)
    
    fmt.Println("\nCircle:")
    printShapeInfo(circle)
    
    // Using interface as a slice
    shapes := []Shape{rect, circle}
    fmt.Println("\nAll shapes:")
    for i, shape := range shapes {
        fmt.Printf("Shape %d: ", i+1)
        printShapeInfo(shape)
    }
}

Embedded Structs (Composition)

Go doesn't have inheritance, but it has composition through embedding:

package main

import "fmt"

// Base struct
type Person struct {
    Name string
    Age  int
}

func (p Person) Introduce() {
    fmt.Printf("Hi, I'm %s and I'm %d years old.\n", p.Name, p.Age)
}

// Employee embeds Person
type Employee struct {
    Person         // embedded struct
    EmployeeID int
    Department string
}

func (e Employee) Work() {
    fmt.Printf("%s is working in %s department.\n", e.Name, e.Department)
}

func main() {
    emp := Employee{
        Person: Person{
            Name: "Alice",
            Age:  30,
        },
        EmployeeID: 12345,
        Department: "Engineering",
    }
    
    // Can access Person's fields directly
    fmt.Println("Name:", emp.Name)
    fmt.Println("Employee ID:", emp.EmployeeID)
    
    // Can call Person's methods
    emp.Introduce()
    
    // Call Employee's methods
    emp.Work()
}

Polymorphism with Interfaces

package main

import "fmt"

// Animal interface
type Animal interface {
    Speak() string
    Move() string
}

// Dog struct
type Dog struct {
    Name string
}

func (d Dog) Speak() string {
    return "Woof!"
}

func (d Dog) Move() string {
    return "Running on four legs"
}

// Bird struct
type Bird struct {
    Name string
}

func (b Bird) Speak() string {
    return "Chirp!"
}

func (b Bird) Move() string {
    return "Flying in the sky"
}

// Function that works with any Animal
func describeAnimal(a Animal) {
    fmt.Printf("Sound: %s, Movement: %s\n", a.Speak(), a.Move())
}

func main() {
    dog := Dog{Name: "Buddy"}
    bird := Bird{Name: "Tweety"}
    
    fmt.Println("Dog:")
    describeAnimal(dog)
    
    fmt.Println("\nBird:")
    describeAnimal(bird)
    
    // Polymorphism in action
    animals := []Animal{dog, bird}
    fmt.Println("\nAll animals:")
    for _, animal := range animals {
        describeAnimal(animal)
    }
}
OOP in Go: Go achieves object-oriented programming through:
  • Encapsulation - Using structs and methods
  • Composition - Embedding structs (instead of inheritance)
  • Polymorphism - Using interfaces
ðŸ’¡ Best Practice: Use pointer receivers for methods that need to modify the struct or when the struct is large. Use value receivers for small structs that don't need modification.
